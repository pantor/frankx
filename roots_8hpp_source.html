43  roots.insert(0.0);
44

46  d = c;
47  c = b;
48  b = a;
49  a = 0.0;

276  if (fl == 0.0) {
277  return l;
278  }
279  if (fh == 0.0) {
280  return h;
281  }

284  xh = h;
285  } else {

287  xl = h;
288  }

291  double dxold = fabs(xh - xl);

295  double temp;
296  for (int j = 0; j < maxIts; j++) {
297  if ((((rts - xh) * df - f) * ((rts - xl) * df - f) > 0.0) ||
298  (fabs(2.0 * f) > fabs(dxold * df))) {
299  dxold = dx;

304  }
305  } else {
